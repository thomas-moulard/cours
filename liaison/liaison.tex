\section{Point to Point Protocol}
\label{sec:PPP}

\begin{itemize}
\item Liaison point à point
\item HDLC ne peut être utilisé dans un environnement multiprotocolé,
  PPP remédie a cet inconvénient
\item Un champ spécifique ``Protocol-ID'' est inséré entre le champ
  ``commande'' et le champ ``données''
\end{itemize}

\begin{figure}[h]
  \centering
\begin{tabular}[c]{| c | c | c | c | c | c | c |}
  \hline
  8 & 8 & 8 & 16 & 16 & 8 & 8 \\
  \hline
  Fanion & Adresse & Commande & Protocole & Données & FCS & Fanion \\
  \hline
  0111 1110 & 1111 1111 &     &           &         &     & 0111 1110 \\
  \hline
\end{tabular}

  \caption{Trame PPP}
  \label{ppp}
\end{figure}

\begin{itemize}
\item Adresse: invisible dans une liaison point à point, valeur fixée
  à FF
\item Protocole:
  \begin{itemize}
  \item 0x0021 $\rightarrow$ IPV4
  \item 0x800F $ \rightarrow$ IPV6
  \end{itemize}
\end{itemize}

\begin{itemize}
\item peut fonctionner aussi bien sur une liaison asynchrone que sur
  une liaison synchrone
\item sur une liaison synchrone $=$  transparence comme dans HDLC
  (insertion d'un zéro de transparence)
\item sur une liaison asynchrone, la transparence du fanion est
  assurée par la séquence de caractère ESC\_PPP, ESC\_FLAG
\item le caractère ESC\_PPP, est remplacé par la séquence ESC\_PPP, ESC\_ESC
\end{itemize}

\section{Serial Line Internet Protocol}
\label{sec:SLIP}

\begin{itemize}
\item asynchrone orienté caractère
\item très simple
\item n'effectue que la délimitation des trames
\item n'offre aucun service complémentaire: pas de détection, ni de
  correction d'erreurs
\item seule la transparence aux caractères de délimitation est réalisée
\end{itemize}

\begin{figure}[h!]
  \centering
  \begin{tabular}[h]{| c | c | c |}
    \hline
    END & Caractères transports & END \\
    0xC0 &                      & 0xC0 \\
    \hline
  \end{tabular}
  \caption{Structure de la trame}
\end{figure}

\begin{itemize}
\item La transparence des caractères END est assurée par la séquence
  ESC\_SLIP, ESC\_END
\item le caractère d'échappement ESC\_SLIP est lui-même remplacé par
  ESC\_SLIP, ESC\_ESC
\end{itemize}

\end{itemize}


\subsection{Trame (I)}

\vspace{0.5cm}

\begin{figure}[h!]
  \centering
  \begin{tabular}[h]{| c | c | c | c |}
    \hline
    0 & $N(S)$ & F/F & $N(R)$ \\
    \hline
  \end{tabular}
\end{figure}

\begin{itemize}
\item 0: code d'identification de trame
\item $N(S)$: numéro de séquence de trame envoyée (s: send)
\item $N(R)$: numéro de séquence de trame reçue (r: receive)
\item Codé sur 3 bits en MOD8: trames numérotées de 0 à 7
\item Codé sur 7 bits en MOD128: trame numérotées de 0 à 127
\end{itemize}

\subsection{Trame de supervision (S)}

\begin{figure}[h!]
  \centering

  \begin{tabular}[b]{| c | c | c | c |}
    \hline
    10 & SS & P/F & $N(R)$ \\
    \hline
  \end{tabular}
\end{figure}

\begin{itemize}
\item 10: code d'identification
\item SS:
  \begin{itemize}
  \item 00: Receive Ready (accusé positif, XON contrôle de flux
  \item 01: Reject (accusé négatif, retransmission séquentielle à
    partir de l'erreur
  \item 10: Receive Not Ready (accusé positif, XOFF contrôle de flux)
  \item 11: Selective REJect (Accusé négatif, retransmission
    uniquement de la trame demandée)
  \end{itemize}
\end{itemize}
Non utilisé dans les WAN.


\subsection{Rappels}

\begin{itemize}
\item Mode ABM: primaire - primaire $\rightarrow$ Full Duplex
\item Mode ARM: primaire - secondaire $\rightarrow$ Half Duplex
\item Mode NRM: maître - esclave
\end{itemize}
